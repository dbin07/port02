let spanSlow = $(".spanSlow");
let spanFast = $(".spanFast");

$(window).on("mousemove",mouseMove);

function mouseMove(e) {
    let position = (e.pageX - ($(window).width()/2))*0.1;     

    spanSlow.css({"transform":"translate("+position+"px)"});
    spanFast.css({"transform":"translate("+-position+"px)"});            
}
